Vueе®һдҫӢеҜ№иұЎ

еҲӣе»әе№¶з»‘е®ҡ Vue еҜ№иұЎ

Vue д№ҹжҳҜдёҖдёӘжһ„йҖ еҮҪж•°,йҖҡиҝҮ new Vue()еҸҜд»ҘеҲӣе»әдёҖдёӘ Vue еҜ№иұЎ,йҖҡиҝҮ Vue еҜ№иұЎиҝӣиЎҢеҜ№ DOM е…ғзҙ д»ҘеҸҠеҶ…йғЁзҡ„еӯҗеӯҷе…ғзҙ зҡ„ж“ҚдҪң

    new Vue({
        el: '#div',
        /*
        еңЁVueдёӯйҖҡиҝҮelеұһжҖ§жқҘз»‘е®ҡдёҖдёӘDOMе…ғзҙ ,д»ҺиҖҢи®©иҜҘDOMе…ғзҙ д»ҘеҸҠеҶ…йғЁйғҪз»‘е®ҡVueзҡ„зӣёе…іж“ҚдҪң,дёҖиҲ¬жҳҜйҖҡиҝҮIDиҝӣиЎҢжҹҘжүҫ,еӣ дёәиҝҷж ·жүҚиғҪеӨҹзІҫзЎ®з»‘е®ҡ
        */
        data: {
            //dataеұһжҖ§дёӯеҢ…еҗ«зқҖеңЁelдёӯдҪҝз”Ёзҡ„дҪҝз”Ёзҡ„еҸҳйҮҸжҲ–еұһжҖ§
            msg: 123,
        },
        //д№ҹеҸҜд»ҘдҪҝз”ЁеҮҪж•°еҪўејҸзҡ„
        data() {
            return {
                msg: 123,
            }
        },
        //methodsеұһжҖ§еҢ…еҗ«зқҖйңҖиҰҒдҪҝз”Ёзҡ„ж–№жі•
        methods: {
            show() {
                console.log(this.msg)
            },
        },
    })

жіЁ:еңЁе®һдҫӢеҶ…йғЁдҪҝз”Ёе®ҡд№үзҡ„еұһжҖ§жҲ–ж–№жі•ж—¶дёҚиғҪзӣҙжҺҘдҪҝз”Ё,еҝ…йЎ»йҖҡиҝҮ this жқҘжҢҮе®ҡйңҖиҰҒз”Ёзҡ„еұһжҖ§
