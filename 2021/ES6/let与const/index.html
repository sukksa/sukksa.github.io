letдёҺconset

еңЁES6дёӯеј•е…ҘдәҶеҸҳйҮҸеЈ°жҳҺзҡ„дёӨз§Қж–№ејҸпјҢlet дёҺ conset

  • let з”ЁжқҘеЈ°жҳҺеҸҳйҮҸ
  • conset з”ЁжқҘеЈ°жҳҺеёёйҮҸ

letе‘Ҫд»Ө

еҹәжң¬з”Ёжі•

ES6 ж–°еўһдәҶletе‘Ҫд»ӨпјҢз”ЁжқҘеЈ°жҳҺеҸҳйҮҸгҖӮе®ғзҡ„з”Ёжі•зұ»дјјдәҺvarпјҢдҪҶжҳҜжүҖеЈ°жҳҺзҡ„еҸҳйҮҸпјҢеҸӘеңЁletе‘Ҫд»ӨжүҖеңЁзҡ„д»Јз Ғеқ—еҶ…жңүж•ҲгҖӮ

    {
        let a = 10
        var b = 1
    }

    a // ReferenceError: a is not defined.
    b // 1

дёҠйқўд»Јз ҒеңЁд»Јз Ғеқ—д№ӢдёӯпјҢеҲҶеҲ«з”Ёletе’ҢvarеЈ°жҳҺдәҶдёӨдёӘеҸҳйҮҸгҖӮз„¶еҗҺеңЁд»Јз Ғеқ—д№ӢеӨ–и°ғз”ЁиҝҷдёӨдёӘеҸҳйҮҸпјҢз»“жһңletеЈ°жҳҺзҡ„еҸҳйҮҸжҠҘй”ҷпјҢvarеЈ°жҳҺзҡ„еҸҳйҮҸиҝ”еӣһдәҶжӯЈзЎ®зҡ„еҖјгҖӮиҝҷиЎЁжҳҺпјҢletеЈ°жҳҺзҡ„еҸҳйҮҸеҸӘеңЁе®ғжүҖеңЁзҡ„д»Јз Ғеқ—жңүж•ҲгҖӮ

дҫӢпјҡйҖҡиҝҮе®ҡж—¶еҷЁжҜҸз§’жү“еҚ°0-4

    for(var i = 0; i < 5; i++){
        setTimeout(function(){
            alert(i)
        },1000)
    }
    // 5 5 5 5 5

еӣ дёәES5д№ӢеүҚжІЎжңүеқ—зә§дҪңз”Ёеҹҹзҡ„жҰӮеҝөпјҢжүҖд»ҘеҫӘзҺҜе®ҡж—¶еҷЁгҖҒеӣһи°ғеҮҪж•°жү§иЎҢж—¶зҡ„ i е®ҡд№үеңЁwindowдёӯпјҢеҪ“еӣһи°ғеҮҪж•°жү§иЎҢж—¶пјҢдјҡеҗ‘дёҠзә§дҪңз”ЁеҹҹдёӯеҜ»жүҫеҲ°window.i жӯӨж—¶ i === 5пјҢж•…

    for(let i = 0; i < 5; i++){
        setTimeout(function(){
            alert(i)
        },1000)
    }
    // 0 1 2 3 4

еңЁеқ—зә§дҪңз”ЁеҹҹдёӯпјҢдҪҝз”Ёletе®ҡд№ү i , зӣёеҪ“дәҺеңЁжҜҸдёӘеқ—зә§дҪңз”Ёеҹҹдёӯе®ҡд№үдәҶдёҖдёӘiпјҢеӣһи°ғеҮҪж•°дёҚеҝ…еңЁе…ЁеұҖдҪңз”ЁеҹҹдёӯеҜ»жүҫiзҡ„еҖј

        for (let i = 0; i < 5; i++) {
            setTimeout(function () {
                alert(i)
            }, 1000)
        } 
        {
            let i = 0
            setTimeout(function () {
                alert(i)
            }, 1000)
        } 
        {
            let i = 1
            setTimeout(function () {
                alert(i)
            }, 1000)
        }

дёҚеӯҳеңЁеҸҳйҮҸжҸҗеҚҮ

varе‘Ҫд»ӨдјҡеҸ‘з”ҹвҖқеҸҳйҮҸжҸҗеҚҮвҖңзҺ°иұЎпјҢеҚіеҸҳйҮҸеҸҜд»ҘеңЁеЈ°жҳҺд№ӢеүҚдҪҝз”ЁпјҢеҖјдёәundefinedгҖӮ
иҖҢletе‘Ҫд»Өж”№еҸҳдәҶиҜӯжі•иЎҢдёәпјҢе®ғжүҖеЈ°жҳҺзҡ„еҸҳйҮҸдёҖе®ҡиҰҒеңЁеЈ°жҳҺеҗҺдҪҝз”ЁпјҢеҗҰеҲҷжҠҘй”ҷгҖӮ

    // var зҡ„жғ…еҶө
    console.log(foo) // иҫ“еҮәundefined
    var foo = 2

    // let зҡ„жғ…еҶө
    console.log(bar) // жҠҘй”ҷReferenceError
    let bar = 2

дёҚе…Ғи®ёйҮҚеӨҚеЈ°жҳҺ

let дёҚе…Ғи®ёеңЁзӣёеҗҢдҪңз”ЁеҹҹеҶ…пјҢйҮҚеӨҚеЈ°жҳҺеҗҢдёҖдёӘеҸҳйҮҸгҖӮvarеҸҜд»Ҙ

    // жҠҘй”ҷ
    function fun1() {
        let a = 10
        var a = 1
    }

    // жҠҘй”ҷ
    function fuc2() {
        let a = 10
        let a = 1
    }

    // жӯЈеёё
    function fu() {
        var a = 10
        var a = 1
    }

consetе‘Ҫд»Ө

еҹәжң¬з”Ёжі•

const еЈ°жҳҺдёҖдёӘеҸӘиҜ»зҡ„еёёйҮҸгҖӮдёҖж—ҰеЈ°жҳҺпјҢеёёйҮҸзҡ„еҖје°ұдёҚиғҪж”№еҸҳпјҢдёҖиҲ¬з”Ё conset еЈ°жҳҺзҡ„еёёйҮҸз”ЁеӨ§еҶҷиЎЁзӨә

    const PI = 3.1415
    PI // 3.1415
    
    PI = 3
    // TypeError: Assignment to constant variable.

constеЈ°жҳҺзҡ„еҸҳйҮҸдёҚеҫ—ж”№еҸҳеҖјпјҢиҝҷж„Ҹе‘ізқҖпјҢconstдёҖж—ҰеЈ°жҳҺеҸҳйҮҸпјҢе°ұеҝ…йЎ»з«ӢеҚіеҲқе§ӢеҢ–пјҢдёҚиғҪз•ҷеҲ°д»ҘеҗҺиөӢеҖј

    const ITEM
    // SyntaxError: Missing initializer in const declaration

constзҡ„дҪңз”ЁеҹҹдёҺletе‘Ҫд»ӨзӣёеҗҢ: еҸӘеңЁеЈ°жҳҺжүҖеңЁзҡ„еқ—зә§дҪңз”ЁеҹҹеҶ…жңүж•ҲгҖӮ

    if (true) {
      const MAX = 5
    }
    console.log(MAX) //ReferenceError: MAX is not defined

consetзҡ„жң¬иҙЁ

  • constе®һйҷ…дёҠдҝқиҜҒзҡ„пјҢе№¶дёҚжҳҜеҸҳйҮҸзҡ„еҖјдёҚеҫ—ж”№еҠЁпјҢиҖҢжҳҜеҸҳйҮҸжҢҮеҗ‘зҡ„йӮЈдёӘеҶ…еӯҳең°еқҖжүҖдҝқеӯҳзҡ„ж•°жҚ®дёҚеҫ—ж”№еҠЁ

  • еҜ№дәҺз®ҖеҚ•зұ»еһӢзҡ„ж•°жҚ®(ж•°еҖјгҖҒеӯ—з¬ҰдёІгҖҒеёғе°”еҖј)пјҢеҖје°ұдҝқеӯҳеңЁеҸҳйҮҸжҢҮеҗ‘зҡ„йӮЈдёӘеҶ…еӯҳең°еқҖпјҢеӣ жӯӨзӯүеҗҢдәҺеёёйҮҸ

  • еҜ№дәҺеӨҚеҗҲзұ»еһӢзҡ„ж•°жҚ®(дё»иҰҒжҳҜеҜ№иұЎе’Ңж•°з»„)пјҢеҸҳйҮҸжҢҮеҗ‘зҡ„еҶ…еӯҳең°еқҖпјҢдҝқеӯҳзҡ„еҸӘжҳҜдёҖдёӘжҢҮеҗ‘е®һйҷ…ж•°жҚ®зҡ„жҢҮй’ҲпјҢconstеҸӘиғҪдҝқиҜҒиҝҷдёӘжҢҮй’ҲжҳҜеӣәе®ҡзҡ„(еҚіжҖ»жҳҜжҢҮеҗ‘еҸҰдёҖдёӘеӣәе®ҡзҡ„ең°еқҖ)пјҢиҮідәҺе®ғжҢҮеҗ‘зҡ„ж•°жҚ®з»“жһ„жҳҜдёҚжҳҜеҸҜеҸҳзҡ„пјҢе°ұе®Ңе…ЁдёҚиғҪжҺ§еҲ¶дәҶ

      const ARR = ["1", "2", "3", "4"]
  ARR.push("5")
  console.log(ARR) //['1', '2', '3', '4', '5']

  ARR = ["a", "b", "c"] //TypeError: Assignment to constant variable.

еҜ№дәҺж•°з»„е’ҢеҜ№иұЎзҡ„е…ғзҙ дҝ®ж”№, дёҚз®—еҒҡеҜ№еёёйҮҸзҡ„дҝ®ж”№, дёҚдјҡжҠҘй”ҷпјҢеӣ дёәconset еҸӘиғҪзЎ®дҝқеҜ№иұЎжүҖжҢҮеҗ‘зҡ„ең°еқҖдёҚиғҪж”№еҸҳпјҢиҖҢең°еқҖзҡ„еҶ…е®№е°ұж— жі•дҝқиҜҒдәҶ

еқ—зә§дҪңз”Ёеҹҹ

дёәд»Җд№ҲйңҖиҰҒеқ—зә§дҪңз”Ёеҹҹ

ES5 еҸӘжңүе…ЁеұҖдҪңз”Ёеҹҹе’ҢеҮҪж•°дҪңз”ЁеҹҹпјҢжІЎжңүеқ—зә§дҪңз”ЁеҹҹпјҢиҝҷеёҰжқҘеҫҲеӨҡдёҚеҗҲзҗҶзҡ„еңәжҷҜгҖӮ

  • з¬¬дёҖз§ҚеңәжҷҜпјҢеҶ…еұӮеҸҳйҮҸеҸҜиғҪдјҡиҰҶзӣ–еӨ–еұӮеҸҳйҮҸгҖӮ

      var tmp = new Date()

  function fun() {
      console.log(tmp)
      if (false) {
          var tmp = "hello world"
      }
  }

  fun(); // undefined

    дёҠйқўд»Јз Ғзҡ„еҺҹж„ҸжҳҜпјҢifд»Јз Ғеқ—зҡ„еӨ–йғЁдҪҝз”ЁеӨ–еұӮзҡ„ tmp еҸҳйҮҸпјҢеҶ…йғЁдҪҝз”ЁеҶ…еұӮзҡ„ tmp еҸҳйҮҸгҖӮдҪҶжҳҜпјҢеҮҪж•°fun()жү§иЎҢеҗҺпјҢиҫ“еҮәз»“жһңдёә undefinedпјҢеҺҹеӣ еңЁдәҺеҸҳйҮҸжҸҗеҚҮпјҢеҜјиҮҙеҶ…еұӮзҡ„ tmp еҸҳйҮҸиҰҶзӣ–дәҶеӨ–еұӮзҡ„ tmp еҸҳйҮҸгҖӮ

  • з¬¬дәҢз§ҚеңәжҷҜпјҢз”ЁжқҘи®Ўж•°зҡ„еҫӘзҺҜеҸҳйҮҸжі„йңІдёәе…ЁеұҖеҸҳйҮҸгҖӮ

      for(var i = 0; i < 5; i++){
      setTimeout(function(){
          alert(i)
      },1000)
  }
  // 5 5 5 5 5

    дёҠйқўд»Јз Ғдёӯiе®ҡд№үдёәе…ЁеұҖеҸҳйҮҸпјҢеҜјиҮҙеӣһи°ғеҮҪж•°жү§иЎҢж—¶пјҢдҪҝз”Ёзҡ„жҳҜе…ЁеұҖдҪңз”Ёеҹҹдёӯзҡ„ i

ES6 зҡ„еқ—зә§дҪңз”Ёеҹҹ

з”Ёletе‘Ҫд»Өж–°еўһдәҶеқ—зә§дҪңз”ЁеҹҹпјҢеӨ–еұӮдҪңз”Ёеҹҹж— жі•иҺ·еҸ–еҲ°еҶ…еұӮдҪңз”ЁеҹҹпјҢеҶ…еұӮдҪңз”ЁеҹҹиғҪиҺ·еҸ–еҲ°еӨ–еұӮдҪңз”ЁеҹҹеЈ°жҳҺзҡ„еҸҳйҮҸ

     {
        let a = "aa" 
        {
            console.log(a) // aa
        }
    }

    {
        console.log(b)  // ReferrenceError
        {
            let b = "bb" 
        }
    }

еҗҢж—¶еҶ…еұӮдҪңз”ЁеҹҹеҸҜд»Ҙе®ҡд№үеӨ–еұӮдҪңз”Ёеҹҹзҡ„еҗҢеҗҚеҸҳйҮҸ

  {
    let a = "Hello World"
    {
      let a = "Hello World"
    }
  }

еқ—зә§дҪңз”ЁеҹҹдёҺеҮҪж•°еЈ°жҳҺ

иҝҗиЎҢзҺҜеўғпјҡES6жөҸи§ҲеҷЁгҖӮ

еҸҜе®һзҺ°зҡ„иЎҢдёәпјҡ

  1. е…Ғи®ёеңЁеқ—зә§дҪңз”ЁеҹҹеҶ…еЈ°жҳҺеҮҪж•°

  2. еҮҪж•°еЈ°жҳҺзұ»дјјдәҺvarпјҢ еҚідјҡжҸҗеҚҮеҲ°е…ЁеұҖдҪңз”ЁеҹҹжҲ–еҮҪж•°дҪңз”Ёеҹҹзҡ„еӨҙйғЁ

  3. еҮҪж•°еЈ°жҳҺиҝҳдјҡжҸҗеҚҮеҲ°жүҖжңүеқ—зә§дҪңз”Ёеҹҹзҡ„еӨҙйғЁ

     function func() {
     console.log('outside')
 }
 (function () {
     // var Func = undefined;
     if (false) {
         function func() {
             console.log('inside')
         }
     }
     func() //Uncaught TypeError: func is not a function
 })() //жҠҘй”ҷ

иҖғиҷ‘еҲ°зҺҜеўғеҜјиҮҙзҡ„е·®ејӮеӨӘеӨ§пјҢеә”иҜҘйҒҝе…ҚеңЁеқ—зә§дҪңз”ЁеҹҹдёӯеЈ°жҳҺеҮҪж•°пјҢеҰӮжһңйңҖиҰҒпјҢеә”иҜҘеҶҷжҲҗеҮҪж•°иЎЁиҫҫејҸзҡ„еҪўејҸпјҢиҖҢдёҚжҳҜеҮҪж•°еЈ°жҳҺиҜӯеҸҘ

ES6зҡ„еқ—зә§дҪңз”Ёеҹҹе…Ғи®ёеЈ°жҳҺеҮҪж•°зҡ„и§„еҲҷеҸӘеңЁдҪҝз”ЁеӨ§жӢ¬еҸ·зҡ„жғ…еҶөдёӢжҲҗз«Ӣпјҡ

    'use strict'
    if (true) {
        function Func1() {} // дёҚжҠҘй”ҷ  
    }
    if (true)
        function Func2() {} // жҠҘй”ҷ

             at top level or inside a block.
