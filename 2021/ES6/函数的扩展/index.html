еҮҪж•°еҸӮж•°зҡ„й»ҳи®ӨеҖј

ES6 е…Ғи®ёз»ҷеҮҪж•°еҸӮж•°иөӢеҖјеҲқе§ӢеҖј

еҹәжң¬з”Ёжі•

йҖҡиҝҮз»ҷеҪўеҸӮиөӢеҲқе§ӢеҖјпјҢе…·жңүй»ҳи®ӨеҖјзҡ„еҸӮж•°, дёҖиҲ¬дҪҚзҪ®иҰҒйқ еҗҺ(жҪңи§„еҲҷ)пјҢеҸӘжңүеҪ“дј е…Ҙзҡ„еҸӮж•°зҡ„еҖјдёәundefinedж—¶жүҚдјҡдҪҝз”Ёй»ҳи®ӨеҖј

    function log(x, y = "World") {
      console.log(x, y)
    }
    
    log("Hello") // Hello World
    log("Hello", "China") // Hello China
    log("Hello", "") // Hello

дҪҝз”ЁеҸӮж•°й»ҳи®ӨеҖјж—¶пјҢеҮҪж•°дёҚиғҪжңүеҗҢеҗҚеҸӮж•°

    // дёҚжҠҘй”ҷпјҢдҪҶжҳҜе…Ҳдј е…Ҙзҡ„еҗҢеҗҚеҸӮж•°дјҡиў«еҗҺйқўиҰҶзӣ–
    function fun(x, x, y) {
        // ...
    }

    // жҠҘй”ҷ
    function fun(x, x, y = 1) {
        // ...
    }
    // SyntaxError: Duplicate parameter name not allowed in this context

дёҺи§Јжһ„иөӢеҖјй»ҳи®ӨеҖјз»“еҗҲдҪҝз”Ё

еҸӮж•°й»ҳи®ӨеҖјеҸҜд»ҘдёҺи§Јжһ„иөӢеҖјзҡ„й»ҳи®ӨеҖјпјҢз»“еҗҲиө·жқҘдҪҝз”Ё

    function fun({ x, y = 5 }) {
      console.log(x, y);
    }
    
    fun({}); // undefined 5
    fun({ x: 1 }); // 1 5
    fun({ x: 1, y: 2 }); // 1 2
    fun(); // TypeError: Cannot read property 'x' of undefined

дёҠйқўд»Јз ҒеҸӘдҪҝз”ЁдәҶеҜ№иұЎзҡ„и§Јжһ„иөӢеҖјй»ҳи®ӨеҖјпјҢжІЎжңүдҪҝз”ЁеҮҪж•°еҸӮж•°зҡ„й»ҳи®ӨеҖј

еҸӘжңүеҪ“еҮҪж•°funзҡ„еҸӮж•°жҳҜдёҖдёӘеҜ№иұЎж—¶пјҢеҸҳйҮҸxе’ҢyжүҚдјҡйҖҡиҝҮи§Јжһ„иөӢеҖјз”ҹжҲҗгҖӮеҰӮжһңеҮҪж•°funи°ғз”Ёж—¶жІЎжҸҗдҫӣеҸӮж•°пјҢеҸҳйҮҸxе’Ңyе°ұдёҚдјҡз”ҹжҲҗпјҢд»ҺиҖҢжҠҘй”ҷ

    function fun({ x, y = 5 } = {}) {
      console.log(x, y)
    }
    
    fun() // undefined 5

дёҠйқўд»Јз ҒжҢҮе®ҡпјҢеҰӮжһңжІЎжңүжҸҗдҫӣеҸӮж•°пјҢеҮҪж•°funзҡ„еҸӮж•°й»ҳи®ӨдёәдёҖдёӘз©әеҜ№иұЎ

дҫӢпјҡ

    function fetch(url, { body = "", method = "GET", headers = {} }) {
      console.log(method);
    }
    
    fetch("https://example.com", {});
    // "GET"
    
    fetch("https://example.com");
    // жҠҘй”ҷ

дёҠйқўд»Јз ҒдёӯпјҢеҰӮжһңеҮҪж•°fetchзҡ„з¬¬дәҢдёӘеҸӮж•°жҳҜдёҖдёӘеҜ№иұЎпјҢе°ұеҸҜд»Ҙдёәе®ғзҡ„дёүдёӘеұһжҖ§и®ҫзҪ®й»ҳи®ӨеҖјгҖӮ
иҝҷз§ҚеҶҷжі•дёҚиғҪзңҒз•Ҙз¬¬дәҢдёӘеҸӮж•°пјҢеҰӮжһңз»“еҗҲеҮҪж•°еҸӮж•°зҡ„й»ҳи®ӨеҖјпјҢе°ұеҸҜд»ҘзңҒз•Ҙз¬¬дәҢдёӘеҸӮж•°гҖӮиҝҷж—¶пјҢе°ұеҮәзҺ°дәҶеҸҢйҮҚй»ҳи®ӨеҖјгҖӮ

    function fetch(url, { body = "", method = "GET", headers = {} } = {}) {
      console.log(method);
    }
    
    fetch("https://example.com");
    // "GET"

дёҠйқўд»Јз ҒдёӯпјҢеҮҪж•°fetchжІЎжңүз¬¬дәҢдёӘеҸӮж•°ж—¶пјҢеҮҪж•°еҸӮж•°зҡ„й»ҳи®ӨеҖје°ұдјҡз”ҹж•ҲпјҢз„¶еҗҺжүҚжҳҜи§Јжһ„иөӢеҖјзҡ„й»ҳи®ӨеҖјз”ҹж•ҲпјҢеҸҳйҮҸmethodжүҚдјҡеҸ–еҲ°й»ҳи®ӨеҖјGET

й—®пјҡд»ҘдёӢд»Јз ҒжңүдҪ•еҢәеҲ«

    // еҶҷжі•дёҖ
    function m1({ x = 0, y = 0 } = {}) {
      return [x, y]
    }
    
    // еҶҷжі•дәҢ
    function m2({ x, y } = { x: 0, y: 0 }) {
      return [x, y]
    }

дёҠйқўдёӨз§ҚеҶҷжі•йғҪеҜ№еҮҪж•°зҡ„еҸӮж•°и®ҫе®ҡдәҶй»ҳи®ӨеҖјпјҢеҢәеҲ«жҳҜеҶҷжі•дёҖеҮҪж•°еҸӮж•°зҡ„й»ҳи®ӨеҖјжҳҜз©әеҜ№иұЎпјҢдҪҶжҳҜи®ҫзҪ®дәҶеҜ№иұЎи§Јжһ„иөӢеҖјзҡ„й»ҳи®ӨеҖј
еҶҷжі•дәҢеҮҪж•°еҸӮж•°зҡ„й»ҳи®ӨеҖјжҳҜдёҖдёӘжңүе…·дҪ“еұһжҖ§зҡ„еҜ№иұЎпјҢдҪҶжҳҜжІЎжңүи®ҫзҪ®еҜ№иұЎи§Јжһ„иөӢеҖјзҡ„й»ҳи®ӨеҖј

    // еҮҪж•°жІЎжңүеҸӮж•°зҡ„жғ…еҶө
    m1() // [0, 0]
    m2() // [0, 0]
    
    // x е’Ң y йғҪжңүеҖјзҡ„жғ…еҶө
    m1({ x: 3, y: 8 }) // [3, 8]
    m2({ x: 3, y: 8 }) // [3, 8]
    
    // x жңүеҖјпјҢy ж— еҖјзҡ„жғ…еҶө
    m1({ x: 3 }) // [3, 0]
    m2({ x: 3 }) // [3, undefined]
    
    // x е’Ң y йғҪж— еҖјзҡ„жғ…еҶө
    m1({}) // [0, 0];
    m2({}) // [undefined, undefined]

еҮҪж•°зҡ„ length еұһжҖ§

жҢҮе®ҡдәҶй»ҳи®ӨеҖјд»ҘеҗҺпјҢеҮҪж•°зҡ„lengthеұһжҖ§пјҢе°Ҷиҝ”еӣһжІЎжңүжҢҮе®ҡй»ҳи®ӨеҖјзҡ„еҸӮж•°дёӘж•°

    (function (a) {}).length // 1
    (function (a = 5) {}).length // 0
    (function (a, b, c = 5) {}).length // 2

еӣ дёәlengthеұһжҖ§зҡ„еҗ«д№үжҳҜпјҢиҜҘеҮҪж•°йў„жңҹдј е…Ҙзҡ„еҸӮж•°дёӘж•°гҖӮжҹҗдёӘеҸӮж•°жҢҮе®ҡй»ҳи®ӨеҖјд»ҘеҗҺпјҢйў„жңҹдј е…Ҙзҡ„еҸӮж•°дёӘж•°е°ұдёҚеҢ…жӢ¬иҝҷдёӘеҸӮж•°дәҶ

еҰӮжһңи®ҫзҪ®дәҶй»ҳи®ӨеҖјзҡ„еҸӮж•°дёҚжҳҜе°ҫеҸӮж•°пјҢйӮЈд№ҲlengthеұһжҖ§д№ҹдёҚеҶҚи®Ўе…ҘеҗҺйқўзҡ„еҸӮж•°дәҶ

    (function (a, b = 1, c) {}).length  // 1

дҪңз”Ёеҹҹ

дёҖж—Ұи®ҫзҪ®дәҶеҸӮж•°зҡ„й»ҳи®ӨеҖјпјҢеҮҪж•°иҝӣиЎҢеЈ°жҳҺеҲқе§ӢеҢ–ж—¶пјҢеҸӮж•°дјҡеҪўжҲҗдёҖдёӘеҚ•зӢ¬зҡ„дҪңз”Ёеҹҹ(context)гҖӮзӯүеҲ°еҲқе§ӢеҢ–з»“жқҹпјҢиҝҷдёӘдҪңз”Ёеҹҹе°ұдјҡж¶ҲеӨұ

    let x = 1
    
    function fun(x, y = x) {
      	console.log(y)
    }
    
    fun(2) // 2

дёҠйқўд»Јз ҒдёӯпјҢеҸӮж•°yзҡ„й»ҳи®ӨеҖјзӯүдәҺxпјҢи°ғз”Ёfun()ж—¶пјҢжӯӨж—¶еҪўжҲҗдәҶе°Ғй—ӯзҡ„дҪңз”ЁеҹҹпјҢй»ҳи®ӨеҖјxжҢҮеҗ‘з¬¬дёҖдёӘеҸӮж•°xпјҢд№ҹе°ұжҳҜ2

    let x = 1

    function fun(y = x) {
        let x = 3
        console.log(y)
    }

    fun() // 1

дёҠйқўд»Јз ҒдёӯпјҢеҮҪж•°fи°ғз”Ёж—¶пјҢеҸӮж•°y = xеҪўжҲҗдёҖдёӘеҚ•зӢ¬зҡ„дҪңз”ЁеҹҹгҖӮиҝҷдёӘдҪңз”ЁеҹҹйҮҢйқўпјҢеҸҳйҮҸxжң¬иә«жІЎжңүе®ҡд№үпјҢжүҖд»ҘжҢҮеҗ‘еӨ–еұӮзҡ„е…ЁеұҖеҸҳйҮҸxгҖӮеҮҪж•°и°ғз”Ёж—¶пјҢеҮҪж•°дҪ“еҶ…йғЁзҡ„еұҖйғЁеҸҳйҮҸxеҪұе“ҚдёҚеҲ°й»ҳи®ӨеҖјеҸҳйҮҸxгҖӮ

еҰӮжһңжӯӨж—¶пјҢе…ЁеұҖеҸҳйҮҸxдёҚеӯҳеңЁпјҢе°ұдјҡжҠҘй”ҷ

rest еҸӮж•°

ES6 еј•е…Ҙ rest еҸӮж•°(еҪўејҸдёәвҖҰеҸҳйҮҸеҗҚ)пјҢз”ЁдәҺиҺ·еҸ–еҮҪж•°зҡ„еӨҡдҪҷеҸӮж•°пјҢиҝҷж ·е°ұдёҚйңҖиҰҒдҪҝз”ЁargumentsеҜ№иұЎдәҶгҖӮ
rest еҸӮж•°жҗӯй…Қзҡ„еҸҳйҮҸжҳҜдёҖдёӘж•°з»„пјҢиҜҘеҸҳйҮҸе°ҶеӨҡдҪҷзҡ„еҸӮж•°ж”ҫе…Ҙж•°з»„дёӯ

    // ES5 иҺ·еҸ–е®һеҸӮзҡ„ж–№ејҸ
    function fun(){
        console.log(arguments)
    }
    fun('a','b','c')

    // rest еҸӮж•°
    function fun(x, ...args){
        console.log(args);// filter some every map 
    }
    fun('a','b','c') // b c

rest еҸӮж•°еҝ…йЎ»иҰҒж”ҫеҲ°еҸӮж•°жңҖеҗҺпјҢеҗҰеҲҷдјҡжҠҘй”ҷ

    function fun(a, ...args, b) {}
    // SyntaxError: Rest parameter must be last formal parameter

еҮҪж•°зҡ„lengthеұһжҖ§пјҢдёҚеҢ…жӢ¬ rest еҸӮж•°

    (function (...args) {}).length // 0
    (function (a, b, ...args) {}).length // 2

з®ӯеӨҙеҮҪж•°

ES6 е…Ғи®ёдҪҝз”ЁвҖңз®ӯеӨҙвҖқ(=>)е®ҡд№үеҮҪж•°

еҹәжң¬з”Ёжі•

    let fun = function (a) {
        return a * a
    }

    // зӯүд»·дәҺ
    let fun = (a) => {
        return a * a
    }

еҰӮжһңз®ӯеӨҙеҮҪж•°еҪ“еҪўеҸӮжңүдё”еҸӘжңүдёҖдёӘзҡ„ж—¶еҖҷпјҢеҸҜд»ҘзңҒз•Ҙ()
еҪ“д»Јз ҒдҪ“еҸӘжңүдёҖжқЎиҜӯеҸҘзҡ„ж—¶еҖҷ, еҸҜд»ҘзңҒз•Ҙ{}пјҢдҪҶжӯӨж—¶ return еҝ…йЎ»зңҒз•ҘпјҢreturn иҝ”еӣһиҜҘжқЎиҜӯеҸҘжү§иЎҢз»“жһң

    let fun = (a) => {
        return a * a
    }

    //зӯүд»·дәҺ
    let fun = a => a * a

з”ұдәҺеӨ§жӢ¬еҸ·иў«и§ЈйҮҠдёәд»Јз Ғеқ—пјҢжүҖд»ҘеҰӮжһңз®ӯеӨҙеҮҪж•°зӣҙжҺҘиҝ”еӣһдёҖдёӘеҜ№иұЎпјҢеҝ…йЎ»еңЁеҜ№иұЎеӨ–йқўеҠ дёҠжӢ¬еҸ·пјҢеҗҰеҲҷдјҡжҠҘй”ҷ

    // жҠҘй”ҷ
    let getTempItem = id => { id: id, name: "Temp" }
    
    // дёҚжҠҘй”ҷ
    let getTempItem = id => ({ id: id, name: "Temp" })

жіЁж„ҸдәӢйЎ№

  • this жҳҜйқҷжҖҒзҡ„. this е§Ӣз»ҲжҢҮеҗ‘еҮҪж•°еЈ°жҳҺж—¶жүҖеңЁдҪңз”ЁеҹҹдёӢзҡ„ this зҡ„еҖј

     function getName() {
     console.log(this.name)
 }

 let getName2 = () => {
     console.log(this.name)
 }

 window.name = "window"
 const obj = {
     name: "obj"
 }

 // call ж–№жі•и°ғз”Ё

    еҰӮеңЁеӣһи°ғеҮҪж•°дёӯдҪҝз”Ё
    let div = document.getElementById(вҖңtest-divвҖқ)
    div.onclick = function () {
    // setTimeout(function () {
    // console.log(this) // жӯӨж—¶thisжҳҜwindow

    setTimeout(() => {
    // жӯӨж—¶thisжҳҜеӨ–йғЁдҪңз”Ёеҹҹdiv
    this.style.background = вҖңredвҖқ
    }, 2000)
    }

  • дёҚиғҪдҪңдёәжһ„йҖ е®һдҫӢеҢ–еҜ№иұЎ,д№ҹе°ұжҳҜиҜҙпјҢдёҚеҸҜд»ҘдҪҝз”Ёnewе‘Ҫд»ӨпјҢеҗҰеҲҷдјҡжҠӣеҮәдёҖдёӘй”ҷиҜҜ

      let Person = (name, age) => {
      this.name = name
      this.age = age
  }
  let me = new Person('tom',30)

  • дёҚиғҪдҪҝз”Ё arguments еҸҳйҮҸ,иҜҘеҜ№иұЎеңЁеҮҪж•°дҪ“еҶ…дёҚеӯҳеңЁгҖӮеҰӮжһңиҰҒз”ЁпјҢеҸҜд»Ҙз”Ё rest еҸӮж•°д»Јжӣҝ

      let fun = () => {

  fun(1, 2)
